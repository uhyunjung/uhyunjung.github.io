## C++
```c++
#include <string>
#include <vector>

using namespace std;

vector<int> solution(int rows, int columns, vector<vector<int>> queries) {
    vector<int> answer;
    
    vector<vector<int>> arr(rows+1, vector<int>(columns+1, 0));
    for(int i=1; i<=rows; i++) {
        for(int j=1; j<=columns; j++) {
            arr[i][j] = (i - 1) * columns + j;
        }
    }
    
    for(auto q : queries) { // 반시계 방향으로 덮어쓰기
        int r1 = q[0]; int c1 = q[1]; int r2 = q[2]; int c2 = q[3];
        int temp = arr[r1][c1];
        int minNum = arr[r1][c1];
        
        for(int i=r1; i<r2; i++) {
            minNum = min(minNum, arr[i][c1]);
            arr[i][c1] = arr[i+1][c1];
        }
        
        for(int i=c1; i<c2; i++) {
            minNum = min(minNum, arr[r2][i]);
            arr[r2][i] = arr[r2][i+1];
        }
        
        for(int i=r2; i>r1; i--) {
            minNum = min(minNum, arr[i][c2]);
            arr[i][c2] = arr[i-1][c2];
        }

        for(int i=c2; i>c1; i--) {
            minNum = min(minNum, arr[r1][i]);
            arr[r1][i] = arr[r1][i-1];
        }
        
        arr[r1][c1+1] = temp;
        answer.push_back(minNum);
    }
    
    return answer;
}
```
