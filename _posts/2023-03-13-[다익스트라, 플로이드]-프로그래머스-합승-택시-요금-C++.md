## 다익스트라
```c++
#include <string>
#include <vector>
#include <queue>

const int INF = (int)1e8;

using namespace std;

int solution(int n, int s, int a, int b, vector<vector<int>> fares) {
    int answer = INF;
    int dist[201], total[201];
    fill(total, total+201, 0);
    
    vector<pair<int, int>> v[201];
    for(auto f : fares) {
        v[f[0]].push_back({f[1], f[2]});
        v[f[1]].push_back({f[0], f[2]});
    }
    
    vector<int> end;
    end.push_back(s);
    end.push_back(a);
    end.push_back(b);
    
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<>> pq;
    for(auto e : end) {
        fill(dist, dist+201, INF);
        pq.push({0, e});
        dist[e] = 0;
        while(!pq.empty()) {
            auto now = pq.top();
            pq.pop();
            
            if (dist[now.second] < now.first) continue;

            for(auto vec : v[now.second]) {
                if (dist[vec.first] < now.first + vec.second) continue;
                dist[vec.first] = now.first + vec.second;
                pq.push({dist[vec.first], vec.first});
            }
        }
        
        for(int i=1; i<=n; i++) total[i] += dist[i];
    }
    
    for(int i=1; i<=n; i++) answer = min(answer, total[i]);
    
    return answer;
}
```

## 플로이드
```c++
#include <string>
#include <vector>
#include <queue>

const int INF = (int)1e8;

using namespace std;

int solution(int n, int s, int a, int b, vector<vector<int>> fares) {
    int answer = INF;
    int g[n+1][n+1];
    
    for(int i=1; i<=n; i++) {
        for(int j=1; j<=n; j++) {
            if (i == j) g[i][j] = 0;
            else g[i][j] = INF;
        }
    }
    for(auto f : fares) {
        g[f[0]][f[1]] = f[2];
        g[f[1]][f[0]] = f[2];
    }

    for(int k=1; k<=n; k++) {
        for(int i=1; i<=n; i++) {
            for(int j=1; j<=n; j++) {
                g[i][j] = min(g[i][j], g[i][k] + g[k][j]);
            }
        }
    }
    
    for(int i=1; i<=n; i++)  answer = min(answer, g[s][i] + g[i][a] + g[i][b]);
    
    return answer;
}
```
