## C++

```c++
#include <vector>

using namespace std;

int n, m;
int dx[4] = {0, 0, 1, -1};
int dy[4] = {1, -1, 0, 0};
vector<vector<int>> Board;

int dfs(vector<int> now, vector<int> next){
    if(!Board[now[0]][now[1]]) return 0;
    
    bool isPossible = false; // 한 번이라도 승리하면 true
    int maxMove = 0;
    int minMove = m*n;
    for(int i = 0; i < 4; i++){
        int newX = now[0] + dx[i];
        int newY = now[1] + dy[i];
        if ((newX < 0) || (newX >= n) || (newY < 0) || (newY >= m)) continue;
        if (!Board[newX][newY]) continue;
        
        Board[now[0]][now[1]] = 0;
        int val = dfs(next, {newX, newY}) + 1;
        Board[now[0]][now[1]] = 1;

        if(val & 1) { // 승리
            isPossible = true;
            minMove = min(minMove, val);
        }
        else if(!isPossible) maxMove = max(maxMove, val); // 패배
    }
    return (isPossible? minMove : maxMove);
}

int solution(vector<vector<int>> board, vector<int> aloc, vector<int> bloc) {
    n = board.size();
    m = board[0].size();
    Board = board;
    return dfs(aloc, bloc);
}
```
