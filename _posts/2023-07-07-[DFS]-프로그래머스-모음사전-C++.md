## C++
```c++
#include <string>
#include <vector>

using namespace std;

int answer, cnt;
string word;
string arr[5] = {"A", "E", "I", "O", "U"};

void dfs(int depth, string str) {
    if (str == word) {
        answer = cnt;
        return;
    }
    cnt++; // 위치 조심
    if (depth == 5) return; // 5자

    for(int i=0; i<5; i++) dfs(depth + 1, str + arr[i]); // 중복 순열
}

int solution(string w) {
    word = w;
    dfs(0, "");
    return answer;
}
```
