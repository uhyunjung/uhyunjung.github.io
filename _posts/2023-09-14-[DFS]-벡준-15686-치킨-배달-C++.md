## 풀이
1. M개 치킨집 선택(조합으로 시간 개선)
2. 치킨집과 집 거리 계산 or BFS로 치킨집과 모든 지도상 거리 계산 후 집과의 거리 확인

## C++
```c++
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int n, m, answer;
vector<pair<int, int>> chicken;
vector<pair<int, int>> home;
bool visited[14];

void dfs(int idx, int depth) {
    if (depth == m) {
        int total = 0;
        for(int i=0; i<home.size(); i++) {
            int dist = 1e9;
            for(int j=0; j<chicken.size(); j++) {
                if (!visited[j]) continue;
                dist = min(dist, abs(home[i].first - chicken[j].first) + abs(home[i].second - chicken[j].second));
            }
            if (dist != 1e9) total += dist;
        }

        answer = min(answer, total);

        return;
    }

    for(int i=idx; i<chicken.size(); i++) {
        if (visited[i]) continue;
        visited[i] = true;
        dfs(i+1, depth+1);
        visited[i] = false;
    }
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);

    cin >> n >> m;

    int num;
    for(int i=0; i<n; i++) {
        for(int j=0; j<n; j++) {
            cin >> num;
            if (num == 2) chicken.push_back({i, j});
            else if (num == 1) home.push_back({i, j});
        }
    }

    answer = 1e9;
    dfs(0, 0);

    cout << answer;

    return 0;
}
```
