## 풀이
1. 가장 적은 어항에 물고기 한 마리씩 넣기
2. 공중 부양 + 1 (주요 인덱스 r*c-c, r*c, n)
3. 물고기 수 조절
4. 어항 일렬 배치
5. 공중 부양 + n/2
6. 어항 일렬 배치
7. 물고기 수 최대, 최소 계산

## C++
```c++
#include <iostream>

using namespace std;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);

    int n, k;
    cin >> n >> k;

    int fish[101][101];
    for(int i=0; i<n; i++) cin >> fish[0][i];

    int answer = 0;
    int maximum = 0;
    for(int i=0; i<n; i++) maximum = max(maximum, fish[0][i]);

    int minimum = 1e9;
    for(int i=0; i<n; i++) minimum = min(minimum, fish[0][i]);

    // 어항 정리
    while (maximum - minimum > k) {
        answer++;

        // 1. 가장 적은 어항에 물고기 한 마리씩 넣기
        for(int i=0; i<n; i++) {
            if (fish[0][i] == minimum) fish[0][i]++;
        }

        // 2. 공중 부양 + 1 (주요 인덱스 r*c-c, r*c, n)
        int r = 1; // 쌓인 어항 세로 길이
        int c = 1; // 쌓인 어항 가로 길이
        
        while (n - r * c >= r) {
            for(int i=0; i<r; i++) {
                for(int j=0; j<c; j++) {
                    fish[c - j][r * c + i] = fish[i][r * c - c + j];
                    fish[i][r * c - c + j] = 0; // 초기화
                }
            }

            int temp = r; // 임시 변수
            r = c + 1;
            c = temp;
        }

        // 3. 물고기 수 조절
        int arr[101][101];
        copy(&fish[0][0], &fish[n-1][n], &arr[0][0]);

        int dx[4] = {1, 0, -1, 0};
        int dy[4] = {0, 1, 0, -1};

        for(int i=0; i<r; i++) {
            for(int j=r*c-c; j<n; j++) {
                if (arr[i][j] == 0) continue;
                for(int p=0; p<4; p++) {
                    int y = i + dy[p];
                    int x = j + dx[p];
                    if ((x < 0) || (y < 0) || (x >= n) || (y >= n)) continue;
                    if (arr[y][x] == 0) continue;

                    int d = (arr[i][j] - arr[y][x]) / 5;
                    if (d > 0) {
                        fish[i][j] -= d;
                        fish[y][x] += d;
                    }
                }
            }
        }

        // 4. 어항 일렬 배치
        int idx = 0;
        for(int i=r*c-c; i<n; i++) {
            for(int j=0; j<r; j++) {
                if (fish[j][i] != 0) {
                    fish[0][idx++] = fish[j][i];
                    if (j != 0) fish[j][i] = 0;
                }
            }
        }

        // 5. 공중 부양 + n/2
        for(int i=0; i<n/2; i++) {
            fish[1][n/2 + i] = fish[0][n/2 - i - 1];
            fish[0][n/2 - i - 1] = 0;
        }

        for(int i=0; i<n/4; i++) {
            fish[2][n/4 * 3 + i] = fish[1][n/4 * 3 - i - 1];
            fish[3][n/4 * 3 + i] = fish[0][n/4 * 3 - i - 1];
            fish[1][n/4 * 3 - i - 1] = 0;
            fish[0][n/4 * 3 - i - 1] = 0;
        }

        copy(&fish[0][0], &fish[n-1][n], &arr[0][0]);

        for(int i=0; i<4; i++) { // 최대 4층
            for(int j=n/4*3; j<n; j++) { // 주요 인덱스(n/4*3, n)
                if (arr[i][j] == 0) continue;
                for(int p=0; p<4; p++) {
                    int y = i + dy[p];
                    int x = j + dx[p];
                    if ((x < 0) || (y < 0) || (x >= n) || (y >= n)) continue;
                    if (arr[y][x] == 0) continue;

                    int d = (arr[i][j] - arr[y][x]) / 5; // abs X
                    if (d > 0) {
                        fish[i][j] -= d;
                        fish[y][x] += d;
                    }
                }
            }
        }

        // 6. 어항 일렬 배치
        idx = 0;
        for(int i=n/4*3; i<n; i++) {
            for(int j=0; j<4; j++) {
                if (fish[j][i] != 0) {
                    fish[0][idx++] = fish[j][i];
                    if (j != 0) fish[j][i] = 0;
                }
            }
        }

        // 7. 물고기 수 최대, 최소 계산
        minimum = 1e9;
        maximum = 0;
        for(int i=0; i<n; i++) {
            minimum = min(minimum, fish[0][i]);
            maximum = max(maximum, fish[0][i]);
        }
    }

    cout << answer;

    return 0;
}
```
