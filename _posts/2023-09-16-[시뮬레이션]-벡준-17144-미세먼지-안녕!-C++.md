## 풀이
1. 미세먼지 동시 확산
2. 시계, 반시계 방향 처리

## C++
```c++
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int r, c, t;
int dx[4] = {1, 0, -1, 0};
int dy[4] = {0, 1, 0, -1};
int a[51][51];
vector<int> machine;

void operate() {
    int temp[51][51];
    fill(&temp[0][0], &temp[50][51], 0);

    for(int i=0; i<r; i++) {
        for(int j=0; j<c; j++) {
            int d = a[i][j];
            if (d <= 0) continue;

            int cnt = 0;
            for(int k=0; k<4; k++) {
                int newY = i + dy[k];
                int newX = j + dx[k];
                if ((newX < 0) || (newY < 0) || (newX >= c) || (newY >= r)) continue;
                if (a[newY][newX] == -1) continue;
                temp[newY][newX] += d/5;
                cnt++;
            }
            temp[i][j] += d - (d/5 * cnt);
        }
    }

    copy(&temp[0][0], &temp[50][51], &a[0][0]);

    // 반시계 방향
    for(int i=machine[0]; i>0; i--) a[i][0] = a[i-1][0];
    for(int i=0; i<c-1; i++) a[0][i] = a[0][i+1];
    for(int i=0; i<machine[0]; i++) a[i][c-1] = a[i+1][c-1];
    for(int i=c-1; i>0; i--) a[machine[0]][i] = a[machine[0]][i-1];
    a[machine[0]][0] = -1;
    a[machine[0]][1] = 0;

    // 시계 방향
    for(int i=machine[1]; i<r; i++) a[i][0] = a[i+1][0];
    for(int i=0; i<c-1; i++) a[r-1][i] = a[r-1][i+1];
    for(int i=r-1; i>machine[1]; i--) a[i][c-1] = a[i-1][c-1];
    for(int i=c-1; i>0; i--) a[machine[1]][i] = a[machine[1]][i-1];
    a[machine[1]][0] = -1;
    a[machine[1]][1] = 0;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);

    cin >> r >> c >> t;

    for(int i=0; i<r; i++) {
        for(int j=0; j<c; j++) {
            cin >> a[i][j];
            if (a[i][j] == -1) machine.push_back(i);
        }
    }

    for(int i=0; i<t; i++) operate();
    
    int answer = 0;
    for(int i=0; i<r; i++) {
        for(int j=0; j<c; j++) {
            if (a[i][j] != -1) answer += a[i][j];
        }
    }

    cout << answer;

    return 0;
}
```
