## C++
```c++
#include <iostream>
#include <unordered_map>
#include <algorithm>

using namespace std;

int answer;
int input[10];
int horse[4];
int flag[4]; // 지름길 위 확인(지름길 : 1, 25 지난 지름길 : 2)
unordered_map<int, int> path; // 지름길

void dfs(int depth, int score) {
    if (depth == 10) {
        answer = max(answer, score);
        return;
    }

    for(int i=0; i<4; i++) {
        if (horse[i] > 40) continue; // 도착한 말 패스
        if (horse[i] > 0 && horse[i] < 40 && horse[i] % 10 == 0 && flag[i] == 0) flag[i] = 1; // 지름길 여부

        int h = horse[i];
        int f = flag[i];
        int cnt = input[depth];

        if (flag[i] == 0) horse[i] += (cnt * 2);
        else {
            while (cnt > 0) {
                if (horse[i] == 25) flag[i] = 2;
                if (flag[i] == 2) break;
                
                horse[i] = path[horse[i]];
                cnt--;
            }
            horse[i] += (5 * cnt);
        }
        
        if (horse[i] > 40) dfs(depth+1, score); // 도착한 경우
        else {
            bool isPossible = true;
            for(int j=0; j<4; j++) {
                if (i == j) continue;
                if (horse[j] > 40) continue; // 도착한 말 패스
                if (horse[i] == horse[j]) { // 말 위치 중복
                    isPossible = false;
                    break;
                }
            }
            
            if (isPossible) dfs(depth+1, score + horse[i]); // 말 이동 가능
        }

        horse[i] = h; // 백트래킹
        flag[i] = f;
    }
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0); cout.tie(0);

    path[10] = 13; path[13] = 16; path[16] = 19; path[19] = 25;
    path[20] = 22; path[22] = 24; path[24] = 25;
    path[30] = 28; path[28] = 27; path[27] = 26; path[26] = 25; // 30 조심

    for(int i=0; i<10; i++) cin >> input[i];

    dfs(0, 0);

    cout << answer;

    return 0;
}
```
